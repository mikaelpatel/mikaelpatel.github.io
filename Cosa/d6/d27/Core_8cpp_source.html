304 USB_Send(uint8_t ep, const void* d, int len)
305 {
306  if (!_usbConfiguration) return (-1);
307 
308  int res = len;
309  const uint8_t* data = (const uint8_t*)d;
310  uint8_t timeout = 250;
311  while (len) {
312  uint8_t n = USB_SendSpace(ep);
313  if (n == 0) {
