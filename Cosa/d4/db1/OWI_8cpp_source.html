187  uint8_t data = 0;

190  switch (m_pin->read(2)) {
191  case 0b00: // Discrepency between device roms
192  if (pos == last) {

196  }
197  else if (pos > last) {

199  next = pos;
200  }
201  else if (m_rom[i] & (1 << j)) {
202  m_pin->write(1, 1);
203  data |= 0x80;
204  }
205  else {
206  m_pin->write(0, 1);
207  next = pos;
208  }
209  break;

211  m_pin->write(1, 1);

213  break;

215  m_pin->write(0, 1);
216  break;
217  case 0b11: // No device detected
