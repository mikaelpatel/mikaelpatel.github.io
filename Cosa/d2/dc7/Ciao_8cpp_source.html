228 Ciao::write(uint8_t type, uint16_t count)
229 {
230  // Tag byte contains count[0..7]
231  if (count < 8) {
232  count |= type;
233  }
234 
235  // Else tag byte contains marker. Succeeding byte counter[8..255]
236  else if (count < 256) {

244  }
