544 CC3000::simple_link_start(uint8_t src)
545 {
546  // Acquire SPI bus and start message transmission after a short delay
547  spi.acquire(this);
548  spi.begin();
549  DELAY(50);
550 
551  // SPI header with special delay
552  spi.transfer(SPI_OP_WRITE);
553  spi.transfer(0);
554  spi.transfer(5);
555  spi.transfer(0);
556  DELAY(50);
557  spi.transfer(0);
558 
559  // HCI header with start command
560  spi.transfer(HCI_TYPE_CMND);
561  spi.transfer(HCI_CMND_SIMPLE_LINK_START & 0xff);
562  spi.transfer(HCI_CMND_SIMPLE_LINK_START >> 8);
563  spi.transfer(1);
564  spi.transfer(src);
565 
566  spi.end();
567  spi.release();
568 
569  // Wait for initialization
570  delay(1000);
