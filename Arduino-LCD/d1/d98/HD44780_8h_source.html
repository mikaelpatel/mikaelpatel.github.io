396  virtual size_t write(uint8_t c)
397  {
398  // Check for special characters
399  if (c < ' ') {
400  switch (c) {
401  case '\a': // Alert: blink the backlight
402  backlight_off();
403  delay(200);
404  backlight_on();
405  delay(200);
406  return (1);
407  case '\b': // Back-space: move cursor back one step (if possible)
408  set_cursor(m_x - 1, m_y);
409  return (1);
410  case '\f': // Form-feed: clear the display
411  display_clear();
412  return (1);
413  case '\n': // New-line: clear line
414  {
415  uint8_t x, y;
416  set_cursor(0, m_y + 1);
417  get_cursor(x, y);
418  m_io.set_mode(true);
419  for (uint8_t i = 0; i < WIDTH; i++) write(' ');
420  m_io.set_mode(false);
421  set_cursor(x, y);
422  }
423  return (1);
424  case '\r': // Carriage-return: move to start of line
425  set_cursor(0, m_y);
426  return (1);
427  case '\t': // Horizontal tab
428  {
429  uint8_t x = m_x + m_tab - (m_x % m_tab);
430  uint8_t y = m_y + (x >= WIDTH);
431  set_cursor(x, y);
432  }
433  return (1);
434  default:
435  return (0);
436  }
437  }
438 
439  // Write character
440  if (m_x == WIDTH) putchar('\n');
441  m_x += 1;
442  m_io.set_mode(true);
443  m_io.write8b(c);
444  m_io.set_mode(false);
445  return (c & 0xff);
446  }
