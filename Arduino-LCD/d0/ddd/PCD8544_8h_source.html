313  virtual size_t write(uint8_t c)
314  {
315  // Check for special characters
316  if (c < ' ') {
317  switch (c) {
318  case '\a': // Check for special character: alert
319  m_mode = ~m_mode;
320  return (1);
321  case '\b': // Check for special character: back-space
322  cursor_set(m_x - 1, m_y);
323  write_data(BACKGROUND, FONT_WIDTH);
324  return (1);
325  case '\f': // Check for special character: form-feed
326  display_clear();
327  return (1);
328  case '\n': // Check for line-feed: clear new line
329  cursor_set(0, m_y + 1);
330  write_data(BACKGROUND, FONT_WIDTH * WIDTH);
331  case '\r': // Carriage-return: move to start of line
332  cursor_set(0, m_y);
333  return (1);
334  case '\t': // Check for horizontal tab
335  {
336  uint8_t x = m_x + m_tab - (m_x % m_tab);
337  uint8_t y = m_y + (x >= WIDTH);
338  cursor_set(x, y);
339  }
340  return (1);
341  default:
342  return (0);
343  }
344  }
345 
346  // Check that the character is not clipped
347  if (m_x == WIDTH) write('\n');
348  m_x += 1;
349 
350  // Access font for character width and bitmap
351  uint8_t width = FONT_WIDTH - 1;
352  const uint8_t* fp = m_font + ((c - ' ') * width);
353 
354  // Write character to the display memory and an extra byte
355  do write_data(m_mode ^ pgm_read_byte(fp++)); while (--width);
356  write_data(m_mode);
357  return (1);
